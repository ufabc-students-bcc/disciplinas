SEG DATA BYTE
MIN DATA BYTE
HORA DATA BYTE
CLOCK DATA BYTE

MOVE SEG, 0
MOVE MIN, 0
MOVE HORA, 0

LOOP:
	OUT	F001, HORA
	OUT F002, MIN
	OUT F003, SEG
	
	IN F000, CLOCK
	ADD SEG, 1
	CMP SEG, 61
	JZ SOMAMIN

	JMP LOOP
	
SOMAMIN:
	MOVE SEG, 0
	ADD MIN, 1
	CMP MIN, 61
	JZ SOMAHORA
	JMP LOOP
	
SOMAHORA:
	MOVE MIN, 0
	ADD HORA, 1
	JMP LOOP
